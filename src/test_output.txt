src PlaceA dest PlaceF minimize km
MATCH  p=(src)-[:ROAD*1..100]->(dest)
WHERE src.name = 'PlaceA'
AND dest.name = 'PlaceF'
AND ALL(n in nodes(p) where 1=size(filter(m in nodes(p) WHERE m=n)))RETURN p AS path,
reduce(km=0, r in relationships(p) | km+r.km) AS totalkm
ORDER BY totalkm ASC
LIMIT 1
