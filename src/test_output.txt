src a dest b minimize attr
MATCH  p=(src)-[:ROAD*1..100]->(dest)
WHERE src.name = a
WHERE dest.name = b
RETURN p AS path,
reduce(attr=0, r in relationships(p) | attr+r.attr) AS totalattr
ORDER BY totalattr ASC
LIMIT 1
